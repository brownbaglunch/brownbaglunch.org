```markdown
# Brown Bag Lunch — bblfr

A static site built with [Hugo](https://gohugo.io) listing speakers, talks, and cities
for the Brown Bag Lunch community. Anyone can invite a speaker to give a free talk during
their team's lunch break — the host provides the food, the speaker provides the knowledge.

## How it works

- **Hosts** browse speakers and talks, then contact a speaker directly via email.
- **Speakers** register by opening a pull request adding their own Markdown file.
- The site is rebuilt and deployed automatically on every merge to `main`.

## Adding or updating your speaker profile

1. Fork this repository.
2. Create a file at `content/speakers/{country}/{city}/{your-slug}.md`
   following the naming rules below.
3. Fill in the front matter (see template below).
4. Open a pull request — once merged, the site rebuilds automatically.

### Naming rules

| Field | Rule | Example |
|-------|------|---------|
| `{country}` | ISO 3166-1 alpha-2 country code, lowercase | `fr`, `gb`, `de` |
| `{city}` | City name slugified (lowercase, diacritics removed, spaces → `-`) | `paris`, `sophia-antipolis` |
| `{your-slug}` | Your full name slugified | `jane-doe.md` |

### Speaker file template

```yaml
---
name: "Jane Doe"
since: "2024-01-15"          # date you joined BBL
city: "fr/paris"             # your home city (ISO country / city slug)
cities:                      # cities where you are available to speak
  - "fr/paris"               # specific city
  - "fr/*"                   # anywhere in France
  - "*/*"                    # anywhere in the world
picture: "https://..."       # URL to your avatar (Gravatar or GitHub avatar recommended)
contacts:
  x: "yourhandle"            # X (formerly Twitter) handle — optional
  mail: "you@example.com"    # email address (required)
websites:
  - name: "GitHub"
    url: "https://github.com/you"
sessions:
  - tags: ["tag1", "tag2"]
    versions:
      - label: "FR"
        flag: "fr"
        title: "Titre de votre talk"
        abstract: |
          Description en **Markdown**.
      - label: "EN"          # optional — add more languages as needed
        flag: "gb"
        title: "Your talk title"
        abstract: |
          Description in **Markdown**.
---

Your bio in Markdown here.
```

### `city` vs `cities`

- `city` — where you **live** (used to place your file in the directory tree).
- `cities` — where you are **willing to speak**. Use `"cc/*"` for "anywhere in country `cc`",
  or `"*/*"` for worldwide availability.

// bblfr_data/scripts/migrate-lib.test.js
const { test } = require('node:test');
const assert = require('node:assert/strict');
const { slugify, toCityPath, convertSession, buildSpeakerMd } = require('./migrate-lib.js');

test('slugify removes diacritics and hyphenates', () => {
  assert.equal(slugify('François Sarradin'), 'francois-sarradin');
  assert.equal(slugify('David Gageot'), 'david-gageot');
  assert.equal(slugify('René Féret'), 'rene-feret');
  assert.equal(slugify('Jean-Pierre Dupont'), 'jean-pierre-dupont');
});

test('toCityPath maps known French cities', () => {
  assert.equal(toCityPath('Paris'), 'fr/paris');
  assert.equal(toCityPath('Lyon'), 'fr/lyon');
  assert.equal(toCityPath('Toulouse'), 'fr/toulouse');
  assert.equal(toCityPath('Sophia Antipolis'), 'fr/sophia-antipolis');
});

test('toCityPath maps known international cities', () => {
  assert.equal(toCityPath('Montréal'), 'ca/montreal');
  assert.equal(toCityPath('Luxembourg'), 'lu/luxembourg');
  assert.equal(toCityPath('Bruxelles'), 'be/bruxelles');
});

test('toCityPath falls back to fr/unknown for unmapped cities', () => {
  assert.equal(toCityPath('XyzUnknown'), 'fr/xyzunknown');
});

test('convertSession creates FR version from fr lang', () => {
  const session = {
    title: 'Mon Talk',
    abstract: 'Contenu du talk',
    tags: ['Java', 'Spring Boot'],
    lang: ['fr'],
  };
  const result = convertSession(session);
  assert.equal(result.versions[0].label, 'FR');
  assert.equal(result.versions[0].flag, 'fr');
  assert.equal(result.versions[0].title, 'Mon Talk');
  assert.equal(result.versions[0].abstract, 'Contenu du talk');
});

test('convertSession creates EN version from en lang', () => {
  const session = { title: 'My Talk', abstract: 'Content', tags: [], lang: ['en'] };
  const result = convertSession(session);
  assert.equal(result.versions[0].label, 'EN');
  assert.equal(result.versions[0].flag, 'gb');
});

test('convertSession lowercases and slugifies tags', () => {
  const session = { title: 'T', abstract: 'A', tags: ['Docker', 'Spring Boot', 'CI/CD'], lang: ['fr'] };
  const result = convertSession(session);
  assert.deepEqual(result.tags, ['docker', 'spring-boot', 'ci-cd']);
});

test('buildSpeakerMd renames twitter to x', () => {
  const speaker = {
    name: 'David Gageot', since: '2012-11-05', bio: 'Dev',
    picture: 'http://...', location: 'Paris', cities: ['Paris'],
    contacts: { twitter: 'dgageot', mail: 'david@gageot.net' },
    websites: [], sessions: [],
  };
  const md = buildSpeakerMd(speaker);
  assert.ok(md.includes('x:'), 'should contain x field');
  assert.ok(!md.includes('twitter:'), 'should not contain twitter field');

const LANG_FLAG = { fr: 'fr', en: 'gb', de: 'de', es: 'es', pt: 'pt', it: 'it' };

function removeDiacritics(str) {
  return str.normalize('NFD').replace(/[\u0300-\u036f]/g, '');
}

function slugify(name) {
  return removeDiacritics(name)
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '');
}

function toCityPath(cityName) {
  if (!cityName) return 'fr/unknown';
  const slug = slugify(cityName.trim());
  const country = CITY_COUNTRY_MAP[slug] || CITY_COUNTRY_MAP[cityName.toLowerCase().trim()] || 'fr';
  return `${country}/${slug}`;
}

function convertSession(session) {
  const primaryLang = (session.lang && session.lang[0]) || 'fr';
  const label = primaryLang.toUpperCase();
  const flag = LANG_FLAG[primaryLang] || primaryLang;
  return {
    tags: (session.tags || []).map(t => slugify(t.trim())),
    versions: [{
      label,
      flag,
      title: session.title || '',
      abstract: session.abstract || '',
    }],
  };
}

function buildSpeakerMd(speaker) {
  // Determine home city
  const locationCity = speaker.location
    ? toCityPath(speaker.location.split(',')[0].trim())
    : null;
  const citiesRaw = speaker.cities || (speaker.location ? [speaker.location.split(',')[0].trim()] : ['unknown']);
  const city = locationCity || toCityPath(citiesRaw[0]);

  // Convert cities array
  const cities = citiesRaw.map(c => toCityPath(c.trim()));

  // Contacts: rename twitter → x
  const contacts = {};
  if (speaker.contacts) {
    if (speaker.contacts.twitter) contacts.x = speaker.contacts.twitter;
    if (speaker.contacts.mail) contacts.mail = speaker.contacts.mail;
  }

  // Collect all unique tags across sessions for Hugo taxonomy
  const allTags = [...new Set(
    (speaker.sessions || []).flatMap(s => (s.tags || []).map(t => slugify(t.trim())))
  )];

  const frontMatter = {
    name: speaker.name,
    since: speaker.since,
    city,
    cities,
    tags: allTags,
    picture: speaker.picture || '',
    contacts,
    websites: (speaker.websites || []).map(w => ({ name: w.name, url: w.url })),
    sessions: (speaker.sessions || []).map(convertSession),
  };

  const yamlStr = yaml.dump(frontMatter, { lineWidth: 120, forceQuotes: false });
  const bio = (speaker.bio || '')
    .replace(/<br\s*\/?>/gi, '\n')
    .replace(/<[^>]+>/g, '')
    .trim();

  return `---\n${yamlStr}---\n\n${bio}\n`;
}

module.exports = { slugify, toCityPath, parseLocation: (l) => l ? toCityPath(l.split(',')[0].trim()) : null, convertSession, buildSpeakerMd };

const vm = require('vm');

let speakerCount = 0;
const slugsSeen = new Set();
for (const speaker of speakers) {
  const cityPath = speaker.location
    ? toCityPath(speaker.location.split(',')[0].trim())
    : toCityPath((speaker.cities || ['unknown'])[0]);
  const [country, city] = cityPath.split('/');
  
  let slug = slugify(speaker.name);
  // Handle duplicate slugs
  if (slugsSeen.has(slug)) {
    slug = `${slug}-2`;
  }
  slugsSeen.add(slug);

  const dir = path.join(BBLFR, 'content', 'speakers', country, city);
  fs.mkdirSync(dir, { recursive: true });
  fs.writeFileSync(path.join(dir, `${slug}.md`), buildSpeakerMd(speaker));
  speakerCount++;
}

// Generate city _index.md files
let cityCount = 0;
for (const loc of (locations || [])) {
  const cityPath = toCityPath(loc.name);
  const [country, city] = cityPath.split('/');
  const dir = path.join(BBLFR, 'content', 'villes', country, city);
  fs.mkdirSync(dir, { recursive: true });
  const [locCountry, locCity] = cityPath.split('/');
  const fm = [
    '---',
    `title: "${loc.name}"`,
    `country: "${locCountry}"`,
    `city: "${locCity}"`,
    `lat: ${loc.lat || 0}`,
    `lng: ${loc.lng || 0}`,
    `image: "${loc.ville_img || ''}"`,
    '---',
    '',
  ].join('\n');
  fs.writeFileSync(path.join(dir, '_index.md'), fm);
  cityCount++;
}
console.log(`✓ Generated ${cityCount} city pages`);

## Task 13: GitHub Actions CI/CD

**Files:**
- Create: `.github/workflows/deploy.yml`

- [ ] **Step 1: Create deploy.yml**

```yaml
# .github/workflows/deploy.yml
name: Build and Deploy

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          submodules: false
          fetch-depth: 0

      - name: Setup Hugo
        uses: peaceiris/actions-hugo@v3
        with:
          hugo-version: '0.157.0'
          extended: true

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: bblfr/package-lock.json

      - name: Install dependencies
        run: cd bblfr && npm ci

      - name: Build
        run: cd bblfr && npm run build
        env:
          HUGO_ENVIRONMENT: production

      - name: Deploy to GitHub Pages
        if: github.ref == 'refs/heads/main'
        uses: peaceiris/actions-gh-pages@v4
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: bblfr/public
```
